RaphaGraph

 
		var rg = new RaphaGraph({
			container:'container',
			type:'pie',
			units: 'slices',
			groups:false,
			aData:[
				[2,2,2,1],
				[5, 2, 1, 1],
				[2, 3, 0, 4]
			],
			w: 600, h: 400,
			names:{cat:['Halloween', 'Thanksgiving', 'Christmas'], var:['Pecan', 'Pumpkin', 'Peanut Butter', 'Apple']},
			title:'Pies I\'ve Eaten This Year',
			fnS: function(){
				var holiday = this.data('cat'),
					filling = this.data('var'),
					slices = this.data('val'),
					s = 'Holiday: '+holiday+'\n';
					if (filling) {
						s += 'Filling: '+filling;
						s += '\n'+'Slices: '+slices;
						if (slices >= 3) {
							s += '\nwhoa, so you like '+filling+'??!!';
						}
					} else {
						s += 'Slices: '+slices;
						if (slices >= 8) {
							s += '\nhungry ehh??';
						}
					}
					return s;
			},
			animationLegnth:750
		});
